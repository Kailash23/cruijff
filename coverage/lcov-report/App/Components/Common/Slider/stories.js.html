class Container extends React.PureComponent {
  state = { value: this.props.initialValue }
 
  handleChange = (value) => {
    this.setState({ value });
  }
 
  render() {
    const {
      initialValue, minimumValue, maximumValue, ...rest
    } = this.props;
    const { value } = this.state;
    return (
      <Block>
        <Text>{value}</Text>
        <Spacer size="M" />

          onValueChange={this.handleChange}
          {...rest}
        />
      </Block>
    );
  }
}
