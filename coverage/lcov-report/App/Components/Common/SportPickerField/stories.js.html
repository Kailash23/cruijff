class Container extends React.PureComponent {
  state = { sport: null }
 
  handleChange = (sport) => {
    this.setState({ sport });
  }
 
  render() {
    const {
      theme,
      label,
      boxed,
      size,
    } = this.props;
    const { sport } = this.state;
 
    return (
      <SportPickerField
        theme={theme}
        label={label}
        boxed={boxed}
        value={sport}
        onChange={this.handleChange}
        size={size}
      />
    );
  }
}
 
Container.propTypes = {
  theme: PropTypes.oneOf(['black', 'white']),
  size: PropTypes.string,
  label: PropTypes.string,
  boxed: PropTypes.bool,
};
 
Container.defaultProps = {
  theme: 'black',
  size: 'M',
  label: '',
  boxed: false,
};
